import os

def process_document(file_path):
    with open(file_path, 'r') as file:
        content = file.read()
    return content.upper()

if __name__ == "__main__":
    doc = "example.txt"
    if os.path.exists(doc):
        print(process_document(doc))
    else:
        print("File not found.")
